1.4 Komut: cp

cp -R ~/files ~/files-backup # tüm dizin alt yapısını ve içeriğini kopyala

cp -s ~/myfiles/file*.txt ~/myfiles2 # tüm file*.txt dosyalarının sembolik
bağlantılarını ~/myfiles2 dizini altına oluşturur. Sembolik bağlantılar,
Windowstaki kısayollar gibidir

cp -R -s ~/myfiles ~/myfiles2 # tüm dizinleri oluşturur ve içindeki tüm
dosyaların sembolik bağlantılarını oluşturur

cp -u * bak # Dizindeki dosyaları eğer kaynak daha yeni ise günceller

cp -v *.txt dir # kopyalama işleminin durumunu gösterir

cp -p file.txt file.txt.bak # sahiplik ve izin bilgileri ile kopyalar
